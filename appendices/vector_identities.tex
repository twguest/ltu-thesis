\begin{equation}\label{eq: vector identity 1}
\curl\left[\curl\boldsymbol{g}\spatial\right] = \grad\left[\divergence\boldsymbol{g}\spatial\right]-\grad^2\boldsymbol{g}\spatial
\end{equation}

\begin{equation}\label{eq: vector identity 2}
    \curl\left[f\spatial g\spatial\right] =
    f\times\left[\curl g\spatial \right] + \left[ \div f\spatial\right]\times g\spatial
\end{equation}

\begin{equation}\label{eq: vector identity 3}
    \grad\2\left[\boldsymbol{f}\spatial\boldsymbol{g}\spatial \right]
    \boldsymbol{f}\spatial \grad\2\boldsymbol{g} + \boldsymbol{g}\spatial \grad\2\boldsymbol{f}
    + 2\grad\boldsymbol{f}\spatial\cdot\boldsymbol{g}\spatial
    \end{equation}
